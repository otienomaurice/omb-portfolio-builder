## Representative Opening Snippet

```
name: Build Windows Portfolio Builder

on:
  workflow_dispatch:
  push:
    branches:
      - main
    tags:
      - "builder-v*"

permissions:
  contents: write

jobs:
  build-windows:
    name: Build Windows installer and portable app
    runs-on: windows-latest

    steps:
      - name: Check out repository
        uses: actions/checkout@v4

      - name: Set up Node.js
        uses: actions/setup-node@v4
        with:
          node-version: "24"

      - name: Set up pnpm
        uses: pnpm/action-setup@v4
        with:
          version: "11.5.3"

      - name: Install desktop builder dependencies
        run: pnpm install --frozen-lockfile

      - name: Build Windows artifacts
```
